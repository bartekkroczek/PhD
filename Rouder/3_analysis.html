# Bootstrapped CI for slope (non-parametric, 1000 resamples)
boot_reg <- function(data, indices) {
  d <- data[indices, ]
  coef(lm(anti ~ pro, data = d))[2]  # Slope only
}
boot_results <- boot(rou_rep, boot_reg, R = 1000)
boot_ci <- boot.ci(boot_results, type = "bca")
